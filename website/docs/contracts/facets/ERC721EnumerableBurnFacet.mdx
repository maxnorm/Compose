## Overview

ERC721EnumerableBurnFacet provides token burn functionality for ERC-721 tokens while maintaining enumeration integrity. It allows safe removal of tokens from the token enumeration system within the diamond using the burn method.

## Usage Example

<ExpandableCode language="solidity" maxLines={20} title="Example Implementation">
{`pragma solidity ^0.8.30;

interface IDiamondCut {
    enum FacetCutAction { Add, Replace, Remove }
    struct FacetCut {
        address facetAddress;
        FacetCutAction action;
        bytes4[] functionSelectors;
    }
    function diamondCut(FacetCut[] calldata _diamondCut, address _init, bytes calldata _calldata) external;
}

contract AddERC721EnumerableBurnFacet {
    IDiamondCut.FacetCut[] private cuts;
    address constant ERC721EnumBurnFacet = 0x123...ABC; // Replace with actual deployed address

    function addERC721EnumerableBurnFacet() external {
        bytes4[] memory selectors = new bytes4[](2);
        selectors[0] = bytes4(keccak256('getStorage()'));
        selectors[1] = bytes4(keccak256('burn(uint256)'));

        cuts.push(IDiamondCut.FacetCut({
            facetAddress: ERC721EnumBurnFacet,
            action: IDiamondCut.FacetCutAction.Add,
            functionSelectors: selectors
        }));

        // Perform the diamond cut
        IDiamondCut(address(this)).diamondCut(cuts, address(0), "");
    }

    function invokeBurn(uint256 tokenId) external {
        ERC721EnumerableBurnFacet(address(this)).burn(tokenId);
    }
}

interface ERC721EnumerableBurnFacet {
    function getStorage() external view returns (address);
    function burn(uint256 _tokenId) external;
}`}
</ExpandableCode>

## Best Practices

<Callout type="tip" title="Best Practice">
- Ensure the ERC-721 enumeration logic is compatible with this facet before integrating into the diamond.
- Enforce access control properly in the diamond to restrict use of the &#x60;burn&#x60; function to authorized parties.
- Call the &#x60;getStorage&#x60; method to confirm storage integrity during upgrades or migrations.
</Callout>

## Security Considerations

<Callout type="warning" title="Security">
Restrict access to &#x60;burn&#x60; to avoid unauthorized token destruction. Validate the token ID exists before burning. Ensure there are no state dependencies or reentrancy risks introduced by integration.
</Callout>
